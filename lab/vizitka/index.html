// Currency Converter Function
function convertCurrency(amount, fromCurrency, toCurrency) {
    // Exchange rates (simplified example)
    const exchangeRates = {
        'USD': { 'EUR': 0.85, 'GBP': 0.73, 'BYN': 2.5 },
        'EUR': { 'USD': 1.18, 'GBP': 0.86, 'BYN': 2.94 },
        'GBP': { 'USD': 1.37, 'EUR': 1.16, 'BYN': 3.42 },
        'BYN': { 'USD': 0.40, 'EUR': 0.34, 'GBP': 0.29 }
    };
    
    // Input validation
    if (typeof amount !== 'number' || amount <= 0) {
        return "Error: Amount must be a positive number";
    }
    
    if (!exchangeRates[fromCurrency] || !exchangeRates[fromCurrency][toCurrency]) {
        return "Conversion rate not available for these currencies";
    }
    
    // Perform conversion
    const rate = exchangeRates[fromCurrency][toCurrency];
    const converted = (amount * rate).toFixed(2);
    
    return amount + " " + fromCurrency + " = " + converted + " " + toCurrency;
}

// Example usage
console.log(convertCurrency(100, 'USD', 'EUR')); // "100 USD = 85.00 EUR"
console.log(convertCurrency(50, 'EUR', 'BYN'));   // "50 EUR = 147.00 BYN"

// Test with error cases
console.log(convertCurrency(-10, 'USD', 'EUR'));  // Error message
console.log(convertCurrency(100, 'USD', 'JPY'));  // Rate not available
